### 智能存储逻辑

```javascript
// 自动检测环境
const isElectron = window.electronAPI !== undefined;

// 保存数据
async function saveData(data) {
    if (isElectron) {
        // Electron: 保存到本地文件
        await window.electronAPI.saveData(data);
    } else {
        // 浏览器：保存到 localStorage
        localStorage.setItem('life_kline_data', JSON.stringify(data));
    }
}

// 加载数据
async function loadData() {
    if (isElectron) {
        // Electron: 从本地文件读取
        return await window.electronAPI.loadData();
    } else {
        // 浏览器：从 localStorage 读取
        return JSON.parse(localStorage.getItem('life_kline_data'));
    }
}
```
